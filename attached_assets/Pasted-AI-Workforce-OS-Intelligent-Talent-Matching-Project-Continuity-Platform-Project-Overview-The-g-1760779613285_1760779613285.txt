from google import genai

def generate_skill_map(description: str) -> dict:
    model = genai.GenerativeModel("gemini-1.5-pro")
    prompt = f"Extract structured skill requirements from this project description:\n{description}"
    result = model.generate_content(prompt)
    return parse_json(result.text)
